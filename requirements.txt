# Star Tracker workspace dependencies
# Use with Python 3.11 conda env (star311) to match system libcamera .so
#
# Setup:
#   conda create -n star311 python=3.11 -y
#   conda activate star311
#   pip install -r requirements.txt
#
# Then symlink system packages that can't be pip-installed:
#   SITE=$(python3 -c "import site; print(site.getsitepackages()[0])")
#   ln -s /usr/lib/python3/dist-packages/libcamera  $SITE/libcamera
#   ln -s /usr/lib/python3/dist-packages/picamera2  $SITE/picamera2
#   ln -s /usr/lib/python3/dist-packages/pykms      $SITE/pykms
#   ln -s /usr/lib/python3/dist-packages/videodev2  $SITE/videodev2
