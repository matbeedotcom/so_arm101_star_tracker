# Star Tracker workspace dependencies — pip-installable subset.
#
# On Raspberry Pi OS Bookworm the recommended setup is a venv that
# inherits system site-packages so picamera2 / libcamera (which can't
# be pip-installed) are visible:
#
#   sudo apt install -y python3-full python3-venv \
#       python3-picamera2 python3-libcamera \
#       python3-astropy python3-aiohttp python3-pil python3-numpy \
#       python3-smbus2
#   python3 -m venv --system-site-packages .venv
#   source .venv/bin/activate
#   pip install -r requirements.txt
#
# Packages listed below are the ones apt doesn't ship (or that pip's
# version is preferable). Everything else — picamera2, libcamera,
# numpy, pillow, astropy, aiohttp, smbus2 — is inherited from the
# system via --system-site-packages.
